// GLOBAL word list let words = []; // Load dictionary on startup fetch("https://raw.githubusercontent.com/erichague76/spelll/main/wordbase.txt") .then(r => r.text()) .then(text => { console.log("Loaded word list:", text.substring(0, 200) + "..."); const raw = text.toLowerCase().trim(); const list = raw.includes("\n") ? raw.split(/\r?\n/) : raw.split(/\s+/); words = list .map(w => w.trim()) .filter(w => /^[a-z]+$/.test(w)); // alphabetic only console.log("Total words loaded:", words.length); }) .catch(err => console.error("Error loading word list:", err)); // Clear UI function clearAll() { lettersEl.value = ""; requiredEl.value = ""; resultsEl.textContent = ""; lettersEl.focus(); } // Solve function function solve() { const letters = lettersEl.value.toLowerCase().trim(); const required = requiredEl.value.toLowerCase().trim(); if (!letters || !required) { resultsEl.textContent = "Enter letters and required letter."; return; } const matches = words.filter(w => w.includes(required) && [...w].every(ch => letters.includes(ch)) ); matches.sort(); if (!matches.length) { resultsEl.textContent = "No matching words found."; return; } // Format into columns like Tkinter const colCount = 6; const colWidth = 12; let out = `${matches.length} matching words:\n\n`; for (let i = 0; i < matches.length; i += colCount) { const row = matches.slice(i, i + colCount); out += row .map(word => (word + " ".repeat(colWidth)).slice(0, colWidth)) .join("") + "\n"; } resultsEl.textContent = out; }
